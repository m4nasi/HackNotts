CREATE TABLE Account(userID INT NOT NULL, username VARCHAR(255) NOT NULL, password VARCHAR(255) NOT NULL, CONSTRAINT pk_userID PRIMARY KEY (userID)) ENGINE = InnoDB;
CREATE TABLE Link(linkID INT NOT NULL, userID INT NOT NULL, allID INT NOT NULL, CONSTRAINT pk_linkID PRIMARY KEY (linkID), CONSTRAINT fk_userID FOREIGN KEY (userID) REFERENCES Account (userID), CONSTRAINT fk_allID FOREIGN KEY (allID) REFERENCES Requirement (allID)) ENGINE = InnoDB;
CREATE TABLE Requirement(allID INT NOT NULL, allergy VARCHAR(255) NOT NULL, CONSTRAINT pk_allID PRIMARY KEY (allID)) ENGINE = InnoDB;
